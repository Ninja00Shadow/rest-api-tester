test "Very Complicated GET" options { repeat: 1, timeout: 20000} {
    GET "http://127.0.0.1:8000/items"

    expect status == 200

    expect body[0]["id"] == 1
    expect body[1]["id"] == 2
    expect body[2]["id"] == 3
    expect body[2]["id"] == 1
    expect body[5]["id"] == 6

}
